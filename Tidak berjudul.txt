<script>
    // Fungsi pindah halaman/step
    function nextStep(currentStep) {
        document.querySelectorAll('.step').forEach(step => step.classList.remove('active'));
        document.getElementById('step' + currentStep).classList.add('active');
    }

    // List teks sindiran lucu pas tombol "dih gamau" diklik
    const rejectTexts = [
        "apa? ga kedengeran jawaban kamu",
        "kayanya ada yang error deh",
        "gadengerr, kamu bilang apaa",
        "apaaaa noraaaaaaaaaaa"
    ];
    let rejectCount = 0;

    function handleRejection() {
        const textElement = document.getElementById('rejection-text');
        if (rejectCount < rejectTexts.length) {
            textElement.innerText = rejectTexts[rejectCount];
            rejectCount++;
        } else {
            textElement.innerText = "Yeee ga bisa! Harus mau pokoknya! 😜";
        }
        
        // Bonus efek: memperkecil ukuran tombol "dih gamau" biar makin susah diklik
        const rejectBtn = document.getElementById('rejectBtn');
        let currentSize = parseFloat(window.getComputedStyle(rejectBtn).fontSize);
        if (currentSize > 8) {
            rejectBtn.style.fontSize = (currentSize - 2) + 'px';
            rejectBtn.style.padding = '8px 15px';
        }
    }

    // Fungsi simulasi loading Scan Hati
    function startScan() {
        const heart = document.getElementById('heartIcon');
        const status = document.getElementById('scanStatus');
        
        // Kunci biar ga diklik dua kali
        heart.onclick = null; 
        heart.classList.add('scanning');
        status.innerText = "SCANNING...";

        // Step text animasi scanning ala video
        setTimeout(() => { status.innerText = "mengecek kadar salting..."; }, 1500);
        setTimeout(() => { status.innerText = "hampir selesai..."; }, 3500);
        
        // Pindah ke halaman sertifikat setelah 5 detik
        setTimeout(() => {
            nextStep(4);
        }, 5000);
    }
</script>
